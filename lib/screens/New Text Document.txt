import 'package:flutter/material.dart';

class AudienceScreen extends StatefulWidget {
  const AudienceScreen({super.key});

  @override
  State<AudienceScreen> createState() => _AudienceScreenState();
}

class _AudienceScreenState extends State<AudienceScreen> {
  final List<String> _segments = [
    'All Users',
    'High Value',
    'New Users',
    'Returning',
    'At Risk',
  ];
  String _selectedSegment = 'All Users';

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);

    return Scaffold(
      body: CustomScrollView(
        slivers: [
          SliverAppBar.large(
            title: const Text('Audience'),
            floating: true,
          ),
          SliverToBoxAdapter(
            child: Padding(
              padding: const EdgeInsets.symmetric(horizontal: 16),
              child: SingleChildScrollView(
                scrollDirection: Axis.horizontal,
                child: Row(
                  children: _segments.map((segment) {
                    final isSelected = _selectedSegment == segment;
                    return Padding(
                      padding: const EdgeInsets.only(right: 8),
                      child: FilterChip(
                        label: Text(segment),
                        selected: isSelected,
                        onSelected: (selected) {
                          setState(() => _selectedSegment = segment);
                        },
                        selectedColor: theme.colorScheme.primary.withValues(alpha: 0.2),
                        checkmarkColor: theme.colorScheme.primary,
                      ),
                    );
                  }).toList(),
                ),
              ),
            ),
          ),
          SliverToBoxAdapter(
            child: Padding(
              padding: const EdgeInsets.all(16),
              child: Card(
                child: Padding(
                  padding: const EdgeInsets.all(20),
                  child: Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Text(
                        'Segment: $_selectedSegment',
                        style: theme.textTheme.titleMedium?.copyWith(
                          fontWeight: FontWeight.bold,
                        ),
                      ),
                      const SizedBox(height: 16),
                      _buildStatRow('Total Users', '12,458'),
                      _buildStatRow('Active (30d)', '8,234'),
                      _buildStatRow('Avg. Age', '28-35'),
                      _buildStatRow('Top Location', 'Mumbai'),
                      _buildStatRow('Engagement Rate', '4.2%'),
                    ],
                  ),
                ),
              ),
            ),
          ),
          SliverList(
            delegate: SliverChildBuilderDelegate(
              (context, index) {
                final users = [
                  {'name': 'Rahul Sharma', 'email': 'rahul@email.com', 'score': '92'},
                  {'name': 'Priya Patel', 'email': 'priya@email.com', 'score': '87'},
                  {'name': 'Amit Kumar', 'email': 'amit@email.com', 'score': '78'},
                  {'name': 'Sneha Gupta', 'email': 'sneha@email.com', 'score': '71'},
                  {'name': 'Vikram Singh', 'email': 'vikram@email.com', 'score': '65'},
                ];
                if (index >= users.length) return null;
                final user = users[index];
                return ListTile(
                  leading: CircleAvatar(
                    backgroundColor: theme.colorScheme.primary.withValues(alpha: 0.2),
                    child: Text(user['name']![0]),
                  ),
                  title: Text(user['name']!),
                  subtitle: Text(user['email']!),
                  trailing: Container(
                    padding: const EdgeInsets.symmetric(horizontal: 10, vertical: 4),
                    decoration: BoxDecoration(
                      color: theme.colorScheme.primary.withValues(alpha: 0.1),
                      borderRadius: BorderRadius.circular(12),
                    ),
                    child: Text(
                      'Score: ${user['score']}',
                      style: TextStyle(
                        color: theme.colorScheme.primary,
                        fontWeight: FontWeight.w600,
                        fontSize: 12,
                      ),
                    ),
                  ),
                );
              },
              childCount: 5,
            ),
          ),
          const SliverToBoxAdapter(child: SizedBox(height: 100)),
        ],
      ),
    );
  }

  Widget _buildStatRow(String label, String value) {
    return Padding(
      padding: const EdgeInsets.symmetric(vertical: 6),
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceBetween,
        children: [
          Text(label, style: TextStyle(color: Colors.grey[400])),
          Text(value, style: const TextStyle(fontWeight: FontWeight.w600)),
        ],
      ),
    );
  }
}
